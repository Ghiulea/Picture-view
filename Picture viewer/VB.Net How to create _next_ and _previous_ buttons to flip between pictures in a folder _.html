    What I meant was, when I load an image to a picturebox using the openfiledialog I want to be able to go through all images in the  folder containing this image using next and previous buttons.

    That would be difficult using the OpenFileDialog, as that dialog provides the apllication with a selected file.  What you want is a selected folder.

    You would use the OpenFileDialog where you wanted to the user to have the option of selecting several files and you were going to use those files as your list of images to display.

    If you use the FolderBrowser dialog instead of the OpenFileDialog then you can get all the image files in the folder into a list and display them on demand. For this example, you need to add checking for a folder with no image files to display.

    Public Class Form1

            Dim Files() As String
        Dim Pointer As Integer

        Private Sub Button1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button1.Click
                If FolderBrowserDialog1.ShowDialog <> Windows.Forms.DialogResult.Cancel Then
                Files = IO.Directory.GetFiles(FolderBrowserDialog1.SelectedPath)
                End If
            Pointer = -1
            End Sub

        Private Sub Button2_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button2.Click
                Do
                Pointer = (Pointer + 1) Mod Files.Count
                Loop Until checkimagetype(Files(Pointer))
                Try
                PictureBox1.Image = Image.FromFile(Files(Pointer))
                Catch ex As Exception
                MessageBox.Show(ex.Message &     " " & Files(Pointer), My.Application.Info.Title, _
                    MessageBoxButtons.OK, MessageBoxIcon.Error)
                End Try
        End Sub

        Private Sub Button3_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button3.Click
                Do
                Pointer -= 1
                    If Pointer < 0 Then Pointer = Files.Count - 1
                Loop Until CheckImageType(Files(Pointer))
                Try
                PictureBox1.Image = Image.FromFile(Files(Pointer))
                Catch ex As Exception
                MessageBox.Show(ex.Message &     " " & Files(Pointer), My.Application.Info.Title, _
                    MessageBoxButtons.OK, MessageBoxIcon.Error)
                End Try
        End Sub

        Private Function CheckImageType(ByVal F As String) As Boolean
            Static Ext() As String = {".jpg", ".jpeg", ".gif", ".bmp", ".png", "ico"}
                For Each S As String In Ext
                    If F.ToLower.EndsWith(S) Then Return True
            Next
            Return False
        End Function

    End     Class

    Another solution is the following...it works with the normal select file dialog...as long as selection is limited to an actual image file this will work just fine...

    Public Class Form1

    Private CurrentPicture As Integer
    Private CurrentDirectoryFiles As New List(Of String)
    Private SupportedImageFileTypes As List(Of String) = New List(Of String) From {".png", ".gif", ".tif", ".jpg"}

    Private Sub Button1_Click(sender As System.Object, e As System.EventArgs) Handles Button1.Click
        'Previous Button
        If CurrentPicture = 0 Then
            CurrentPicture = CurrentDirectoryFiles.Count - 1
        Else
            CurrentPicture -= 1
        End If
        PictureBox1.ImageLocation = CurrentDirectoryFiles(CurrentPicture)
    End Sub

    Private Sub Button2_Click(sender As System.Object, e As System.EventArgs) Handles Button2.Click
        'Next Button
        If CurrentPicture = CurrentDirectoryFiles.Count - 1 Then
            CurrentPicture = 0
        Else
            CurrentPicture += 1
        End If
        PictureBox1.ImageLocation = CurrentDirectoryFiles(CurrentPicture)
    End Sub

    Private Sub Form1_Load(sender As System.Object, e As System.EventArgs) Handles MyBase.Load
        OpenFileDialog1.ShowDialog()
        For Each File As String In FileIO.FileSystem.GetFiles(FileIO.FileSystem.GetParentPath(OpenFileDialog1.FileName))
            If SupportedImageFileTypes.Contains(FileIO.FileSystem.GetFileInfo(File).Extension) Then
                CurrentDirectoryFiles.Add(File)
            End If
        Next
        CurrentPicture = CurrentDirectoryFiles.IndexOf(OpenFileDialog1.FileName)
        PictureBox1.ImageLocation = OpenFileDialog1.FileName
    End Sub

End Class
